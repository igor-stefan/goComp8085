JMP START
VALOR:     DB 10
NBITS:    DB 0
START:     LDA VALOR    
        MOV B,A        
        MVI C,0        
        MVI D,0xF1    
LOOP:   MOV A,D
        ANA B
        JZ PULA
        INR C
PULA:   MOV A,D
        RAL
        MOV D,a
JNC LOOP
        MOV A,C
        STA NBITS
chanGE:        ORG 0XFf1
        HLT
